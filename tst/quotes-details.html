# -*- coding:utf-8 -*-

from scrapy import cmdline
#scrapy crawl itcast （itcast为爬虫名）

import scrapy

#创建一个爬虫类
class ItcastSpider(scrapy.Spider):
    #爬虫名
    name = "itcast"
    #允许爬虫作用范围
    allowd_domains = ["http://www.itcast.cn/"]
    #爬虫其实的url
    start_urls = ["http://www.itcast.cn/channel/teacher.shtml#"]

    def parse(self, response):
        print response.body
